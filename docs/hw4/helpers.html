def transpose(t):
#

Transposes t (aka flips the rows and cols)

  u = {}

def cosine(a, b, c):
#

This is our version of the cosine equation used to find distance

  x1 = ((a ** 2) + (c ** 2) - (b ** 2))/ (2 * c)
  x2 = max(0, min(1, x1))
  y = ((a ** 2) - x2 ** 2) ** .5
  return x2, y
