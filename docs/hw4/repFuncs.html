Adds the row descriptions to their place based on the input file

    

    

      
  rows = copy.deepcopy(rows)
  for j, s in last(rows).items():
    rows[0][j] = str(rows[0][j]) + ":" + str(s)
  rows.pop(len(rows) - 1)
  for n in range(len(rows)):
    row = rows[n]
    if n == 0:
      row[len(row)] = "thingX"
    else:
      u = t["rows"][(len(t["rows"]) - 1) - n + 1]
      row[len(row)] = last(u)
  return DATA(rows)
