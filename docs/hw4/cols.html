    def __init__(self, t):
        if type(t) == list:
            nt = {}
            for item in t:
                nt[len(nt)] = item
            t = nt
        self.names = t
        self.all = {}
        self.x = {}
        self.y = {}
        self.klass = None

        for n in range(len(t)):
            s = t[n]
            col = NUM(n, s) if re.search("^[A-Z]+", s) else SYM(n, s)
            self.all[len(self.all)] = col
            if not re.search("X$", s):
                if re.search("!$", s):
                    self.klass = col
                if re.search("[!+-]$", s):
                    self.y[len(self.y)] = col
                else:
                    self.x[len(self.x)] = col
